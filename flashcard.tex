\newcommand{\card}[3]{
  %front
  \begin{mdframed}[
    backgroundcolor=bggray,
    align=center,
    linecolor=dark,
    linewidth=4pt,
    innerleftmargin=0.02\paperwidth,
    innerrightmargin=0.02\paperwidth,
    innertopmargin=0.3cm,
    innerbottommargin=0.3cm,
  ]
    \linewithlength{0.3\paperwidth}
    \hspace{\fill}
    \hypertarget{#1 front}{}
    \hyperlink{#2 front}{\color{fgwhite}\huge$\leftarrow$} 
    \hyperlink{#1 back}{\color{fgwhite}\huge$\curvearrowright$} 
    \hyperlink{#3 front}{\color{fgwhite}\huge$\rightarrow$} 
    \hspace{\fill}
    \linewithlength{0.3\paperwidth} 
    \cardcontent 
    \begin{center}
      \multido{}{25}{\color{fgwhite}\LARGE$\bigcirc \ $}
    \end{center}
  \end{mdframed}
  \clearpage

  % back
  \begin{mdframed}[
    backgroundcolor=bggray,
    align=center,
    linecolor=dark,
    linewidth=4pt,
    innerleftmargin=0.02\paperwidth,
    innerrightmargin=0.02\paperwidth,
    innertopmargin=0.3cm,
    innerbottommargin=0.3cm,
  ]
    \linewithlength{0.3\paperwidth}
    \hspace{\fill}
    \hypertarget{#1 back}{}
    \hyperlink{#2 front}{\color{fgwhite}\huge$\leftarrow$} 
    \hyperlink{#1 front}{\color{fgwhite}\huge$\curvearrowleft$} 
    \hyperlink{#3 front}{\color{fgwhite}\huge$\rightarrow$} 
    \hspace{\fill}
    \linewithlength{0.3\paperwidth} 
    \cardcontent 
    \begin{center}
      \multido{}{25}{\color{fgwhite}\LARGE$\bigcirc \ $}
    \end{center}
  \end{mdframed}
  \clearpage
}
